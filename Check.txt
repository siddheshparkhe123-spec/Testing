@Override
public NonIndividualCustomerDetails apply(Map<String, String> params) {
    Map<String, Object> userResponse = requestFactory
        .getRequester(ApiCategory.NON_INDIVIDUAL_CUSTOMER_DETAILS)
        .apply(params);

    NonIndividualCustomerDetails details =
        jsonPathMapper.convert(userResponse, NonIndividualCustomerDetails.class);

    // 70-character name matching logic
    String requestName = params.get("customerName");
    String responseName = details.getCustomerName();

    if (requestName != null && responseName != null) {
        String requestPrefix = requestName.length() >= 70
            ? requestName.substring(0, 70)
            : requestName;
        String responsePrefix = responseName.length() >= 70
            ? responseName.substring(0, 70)
            : responseName;

        if (requestPrefix.equalsIgnoreCase(responsePrefix)) {
            details.setCustomerName(requestName);
        }
    }

    return details;
}
